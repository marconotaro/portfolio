Interactive map of remote alpine spots I visited

Author

Marco Notaro

Published

January 8, 2026

Aim

Here I built an interactive map showing remote alpine huts, lakes, and glaciers I visited.

To make geolocation more efficient, retrieved location coordinates are stored in a JSON file.

On the map, markers are color-coded following the Swiss Alpine Club grading system: red for hikes up to T3, blue for hikes up to T6, dark blue for glacier hikes and magenta for alpine lakes.

# use Open-Elevation to get elevation, since Nominatim doesn't have an API for this

def get_elevation(lat, lon):
  """
  Fetch elevation from Open-Elevation API
  """
  
  url = f"https://api.open-elevation.com/api/v1/lookup?locations={lat},{lon}"
  
  try:
    response = requests.get(url, timeout=10)
    if response.ok: ## when response fails (404) doesn't raise any exception
      return response.json()["results"][0]["elevation"]
  except:
    pass ## bad http status
  
  return None # Handles both failures
def get_coordinates(spots, cachefile):
  """
  Retrieve coordinates of alpine spots with caching
  """

  cachefile = Path(cachefile)
  cache = {}
  
  # load cache
  if cachefile.exists():
    with open(cachefile, "r", encoding="utf-8") as f:
      cache = json.load(f)
  
  # fetch not cached spots
  spots_to_fetch = [s for s in spots if s not in cache]
  
  if spots_to_fetch:
    geolocator = Nominatim(user_agent="alpine_spot_finder")
    geocode = RateLimiter(geolocator.geocode, min_delay_seconds=1)
      
    for spot in spots_to_fetch:
      # location = geocode(f"{spot}")
      location = None
      for country in ["Switzerland", "Italy", "France"]:
        location = geocode(f"{spot}, {country}")
        ## search spot in Switzerland, if spot not found continue to Italy
        if location: 
          break

      if location:
        cache[spot] = {
          "Latitude": location.latitude,
          "Longitude": location.longitude,
          "Elevation": round(get_elevation(location.latitude, location.longitude))
        }
      else:
        print(f"Could not find coordinates for: {spot}")
        cache[spot] = {
          "Latitude": None,
          "Longitude": None,
          "Elevation": None
        }
    
    # update cache
    with open(cachefile, "w", encoding="utf-8") as f:
      json.dump(cache, f, indent=2, ensure_ascii=False)
  
  # build dataframe from cache
  data = [{"Name": spot, **cache[spot]} for spot in spots]
  return pd.DataFrame(data)

import folium
import re
from folium.plugins import MarkerCluster, Geocoder

center_lat = df["Latitude"].mean()
center_lon = df["Longitude"].mean()

# initialize map
m = folium.Map(
  location=[center_lat, center_lon],
  zoom_start=7, 
  tiles="OpenStreetMap"
)

markers = MarkerCluster(
  options={
    "maxClusterRadius": 30, # def: 80, smaller less grouping
    "disableClusteringAtZoom": 10  # no clustering at this zoom and higher
  }
).add_to(m)

def get_marker_color(spot):
  if spot["Type"] == "Lake":
    return "purple"
  
  elif spot["Type"] == "City":
    return "orange"
  
  elif spot["Type"] == "Trail Hike":
    if spot["Path"] in ["T2", "T3"]:
      return "red"
    elif spot["Path"] == "T4":
      return "blue"
  
  elif re.search(r"Glacier", spot["Type"]):
    return "darkblue"
  
  return "gray"  # fallback

# Add markers for each spot
for _, spot in df.iterrows():
  folium.Marker(
    location=[spot["Latitude"], spot["Longitude"]],
    popup=spot["Name"],
    tooltip=f"{spot['Name']} <br> Path: {spot['Path']} <br> Elevation: {spot['Elevation']}",
    icon=folium.Icon(color=get_marker_color(spot), icon="home")
  ).add_to(markers)

# Add search bar
Geocoder(
  collapsed=False, 
  position='topright',
  add_marker=False
).add_to(m)

## utils to clean cache file

def clean_cache_file(cache_file = 'data/coordinates.json', spots=None):
  if spots is None:
    if Path(cache_file).exists():
      os.remove(cache_file)
      print(f"cache_file {cache_file} deleted")
  else:
    if isinstance(spots, str):
      spots = [spots]
    with open(cache_file, "r") as f:
      data = json.load(f)
    for hut in spots:
      if hut in data:
        del data[hut]
        print(f"{hut} deleted")
      else:
        print(f"{hut} not found in cache")
    with open(cache_file, "w") as f:
      json.dump(data, f, indent=2, ensure_ascii=False)

def get_elevation_with_delay(hut):
  time.sleep(1)  # need to avoid rate limit
  return round(get_elevation(hut["Latitude"], hut["Longitude"]))
